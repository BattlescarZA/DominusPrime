# Security Enhancements - DominusPrime 0.9.8

## Overview

This document outlines the architecture for three major security enhancements in DominusPrime 0.9.8:

1. **Shell Execution Confirmation**
2. **Tool/Skills Security Framework**
3. **Configurable Security Levels (User-Configurable)**

### 1.6 Integration with Shell Tool

**Modified execute_shell_command:**
```python
async def execute_shell_command(
    command: str,
    timeout: int = 60,
    cwd: Optional[Path] = None,
    bypass_approval: bool = False,  # For system-initiated commands
    context: Optional[ExecutionContext] = None
) -> ToolResponse:
    """
    Execute shell command with security approval.
    """
    # Get security manager
    security_manager = SecurityManager.get_instance()
    
    if not bypass_approval and security_manager.is_enabled():
        # Intercept and analyze
        result = await security_manager.command_interceptor.intercept(
            command=command,
            context=context or ExecutionContext.from_current()
        )
        
        if result.action == InterceptAction.REQUIRE_APPROVAL:
            # Request approval
            approval = await security_manager.approval_handler.request_approval(
                command=command,
                risk=result.risk_assessment,
                channel=context.channel if context else "console",
                user_id=context.user_id if context else "system"
            )
            
            if approval.decision != ApprovalDecision.APPROVED:
                return ToolResponse(
                    content=[
                        TextBlock(
                            type="text",
                            text=f"Command execution denied: {approval.reason}"
                        )
                    ]
                )
        
        elif result.action == InterceptAction.BLOCKED:
            return ToolResponse(
                content=[
                    TextBlock(
                        type="text",
                        text=f"Command blocked by security policy: {result.reason}"
                    )
                ]
            )
    
    # Execute command (existing logic)
    # ... existing execution code ...
    
    # Log execution
    if security_manager.is_enabled():
        await security_manager.execution_logger.log_execution(
            command=command,
            result=execution_result,
            context=context
        )
    
    return execution_result
```

**Tool Wrapper:**
```python
class ExecutionGuard:
    def __init__(
        self,
        permission_manager: PermissionManager,
        rate_limiter: RateLimiter
    ):
        self.permission_manager = permission_manager
        self.rate_limiter = rate_limiter
    
    async def wrap_tool(
        self,
        tool_func: Callable,
        security_profile: ToolRiskProfile
    ) -> Callable:
        """Wrap tool function with security checks."""
        
        async def secured_tool(*args, **kwargs):
            # Get execution context
            context = ExecutionContext.get_current()
            
            # Check permission
            perm_result = await self.permission_manager.check_permission(
                tool_name=security_profile.name,
                user_id=context.user_id,
                session_id=context.session_id
            )
            
            if not perm_result.allowed:
                return ToolResponse(
                    content=[
                        TextBlock(
                            type="text",
                            text=f"Permission denied: {perm_result.reason}"
                        )
                    ]
                )
            
            # Check rate limit
            if security_profile.max_executions_per_hour:
                if not await self.rate_limiter.check_limit(
                    key=f"{context.user_id}:{security_profile.name}",
                    limit=security_profile.max_executions_per_hour,
                    window_seconds=3600
                ):
                    return ToolResponse(
                        content=[
                            TextBlock(
                                type="text",
                                text=f"Rate limit exceeded for {security_profile.name}"
                            )
                        ]
                    )
            
            # Execute tool
            try:
                result = await tool_func(*args, **kwargs)
                
                # Log execution
                await self._log_execution(
                    tool_name=security_profile.name,
                    context=context,
                    success=True
                )
                
                return result
            
            except Exception as e:
                await self._log_execution(
                    tool_name=security_profile.name,
                    context=context,
                    success=False,
                    error=str(e)
                )
                raise
        
        return secured_tool
```

#### 3.3 Security Manager

**Location:** `src/dominusprime/security/manager.py`

```python
class SecurityManager:
    """Central security management."""
    
    _instance: Optional['SecurityManager'] = None
    
    def __init__(self, config: SecurityConfig):
        self.config = config
        self.current_profile = self._load_profile()
        self.command_interceptor = CommandInterceptor(self.current_profile)
        self.permission_manager = PermissionManager(self.current_profile)
        self.execution_guard = ExecutionGuard(self.permission_manager)
        self.execution_logger = ExecutionLogger()
    
    @classmethod
    def get_instance(cls) -> 'SecurityManager':
        """Get singleton instance."""
        if cls._instance is None:
            raise RuntimeError("SecurityManager not initialized")
        return cls._instance
    
    @classmethod
    def initialize(cls, config: SecurityConfig):
        """Initialize security manager."""
        cls._instance = cls(config)
        return cls._instance
    
    def is_enabled(self) -> bool:
        """Check if security features are enabled."""
        return self.config.enabled
    
    async def change_security_level(
        self,
        level: SecurityLevel,
        user_id: str
    ):
        """Change security level and apply new profile."""
        new_profile = SECURITY_PROFILES[level]
        
        # Save to config
        await self._save_profile(level)
        
        # Apply new settings
        self.current_profile = new_profile
        self.command_interceptor.update_config(new_profile)
        self.permission_manager.update_config(new_profile)
        
        # Log change
        await self.execution_logger.log_security_event(
            event_type="security_level_changed",
            user_id=user_id,
            details={
                "new_level": level.value,
                "timestamp": datetime.utcnow().isoformat()
            }
        )
    
    async def get_security_status(self) -> SecurityStatus:
        """Get current security status and statistics."""
        return SecurityStatus(
            level=self.current_profile.level,
            commands_executed_today=await self._count_executions_today(),
            approvals_pending=len(self.command_interceptor.approval_handler.pending_approvals),
            tools_blocked=await self._count_blocked_tools(),
            last_audit_time=await self._get_last_audit_time()
        )
```

def _configure_security_interactive() -> SecurityLevel:
    """Interactive security level selection."""
    click.echo("\n=== Security Configuration ===\n")
    click.echo("Choose a security level that matches your needs:\n")
    
    choices = [
        ("Open", "Minimal restrictions (experienced users only)"),
        ("Relaxed", "Basic safety checks"),
        ("Balanced", "Recommended for most users ⭐"),
        ("Strict", "High security, frequent confirmations"),
        ("Paranoid", "Maximum security, approve everything")
    ]
    
    for i, (name, desc) in enumerate(choices, 1):
        click.echo(f"{i}. {name:10} - {desc}")
    
    selection = click.prompt(
        "\nSelect security level",
        type=click.IntRange(1, 5),
        default=3  # Balanced
    )
    
    level_map = {
        1: SecurityLevel.OPEN,
        2: SecurityLevel.RELAXED,
        3: SecurityLevel.BALANCED,
        4: SecurityLevel.STRICT,
        5: SecurityLevel.PARANOID
    }
    
    selected_level = level_map[selection]
    profile = SECURITY_PROFILES[selected_level]
    
    click.echo(f"\n✓ Selected: {selected_level.value}")
    click.echo(f"  {profile.description}\n")
    
    return selected_level
