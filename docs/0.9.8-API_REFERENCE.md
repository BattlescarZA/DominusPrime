# DominusPrime 0.9.8 - API Reference

## Security System

### SecurityManager

Central coordinator for all security features.

```python
from dominusprime.security.manager import SecurityManager
from dominusprime.security.config import SecurityConfig

# Initialize (singleton)
security = SecurityManager.initialize(config, db_manager)

# Get instance anywhere
security = SecurityManager.get_instance()

# Intercept shell command
result = await security.intercept_shell_command(command, context)

# Request user approval
approved = await security.request_approval(command, risk_assessment, timeout=120)

# Log execution
await security.log_execution(command, context, approved, executed, return_code, stdout, stderr)
```

### RiskAnalyzer

Analyzes command risk levels.

```python
from dominusprime.security.risk_analyzer import RiskAnalyzer

analyzer = RiskAnalyzer(config)

# Analyze command risk
risk_assessment = await analyzer.analyze_risk(command, context)

print(f"Risk Level: {risk_assessment.level}")
print(f"Score: {risk_assessment.score}")
print(f"Reason: {risk_assessment.reason}")
```

### SecurityConfig

Configuration for security system.

```python
from dominusprime.security.config import SecurityConfig
from dominusprime.security.profiles import SecurityLevel

# Create from defaults
config = SecurityConfig(security_level=SecurityLevel.BALANCED)

# Load from file
config = SecurityConfig.from_file("./config/security.json")

# Save to file
config.to_file("./config/security.json")

# Access settings
if config.shell_require_approval:
    # Request approval
    pass
```

### Secure Shell Execution

```python
from dominusprime.agents.tools.shell_secure import execute_shell_command_secure

# Execute with security
return_code, stdout, stderr = await execute_shell_command_secure(
    command="rm file.txt",
    timeout=300,
    working_dir="./project",
    session_id="session123",
    user_id="user456"
)

# With fallback (if security not initialized)
from dominusprime.agents.tools.shell_secure import execute_shell_command_with_fallback

return_code, stdout, stderr = await execute_shell_command_with_fallback(
    command="ls",
    use_security=True  # Falls back to direct execution if not initialized
)
```

## Experience System

### ExperienceSystem

Main coordinator for experience learning and skill extraction.

```python
from dominusprime.agents.memory.experience.system import ExperienceSystem
from pathlib import Path

# Initialize (requires chat_model, formatter, and db_manager)
system = ExperienceSystem(
    working_dir=Path("./working"),
    chat_model=your_chat_model,  # ChatModelBase instance
    formatter=your_formatter,     # FormatterBase instance
    db_manager=db_manager         # DatabaseManager instance
)

# Distill experiences from conversation
await system.distill_experiences(
    messages=conversation_messages,
    session_id="session123"
)

# Generate skills from experiences
generated_skills = await system.generate_skills(session_id="session123")

# Search experiences
results = await system.search_experiences(
    query="python error handling",
    limit=5
)

# Get statistics
stats = await system.get_statistics()
```

### ConversationAnalyzer

Analyzes conversations for learning opportunities.

```python
from dominusprime.agents.memory.experience.analyzer import ConversationAnalyzer

analyzer = ConversationAnalyzer(chat_model=your_chat_model)

# Analyze conversation
analysis = await analyzer.analyze_conversation(messages, session_id)

# Identify learning moments
moments = await analyzer.identify_learning_moments(analysis)

# Access results
for moment in moments:
    print(f"Type: {moment.moment_type}")
    print(f"Description: {moment.description}")
    print(f"Importance: {moment.importance_score}")
```

### SkillGenerator

Generates markdown skills from experiences.

```python
from dominusprime.agents.memory.experience.skill_generator import SkillGenerator

generator = SkillGenerator(
    skills_dir=Path("./working/skills"),
    chat_model=your_chat_model,
    formatter=your_formatter

should_generate = await generator.should_generate_skill(experience)
```

## Multimodal System

### MediaProcessor

Processes media files for storage.

```python
from dominusprime.agents.memory.multimodal.processor import MediaProcessor

processor = MediaProcessor(storage_dir=Path("./data/media"))

# Process local file
media_item = await processor.process_file(
    file_path="image.png",
    session_id="session123",
    user_context="Error screenshot"

media_item = await processor.process_url(
    url="https://example.com/image.png",
    session_id="session123"
)

# Delete media
deleted = await processor.delete_media(media_item)
```

### MultimodalEmbedder

Generates embeddings for multimodal content.

```python
from dominusprime.agents.memory.multimodal.embedder import MultimodalEmbedder

embedder = MultimodalEmbedder(
    visual_model="clip-vit-base-patch32",
    text_model="sentence-transformers/all-MiniLM-L6-v2"
)

# Embed image
visual_embedding = await embedder.embed_image(media_item)

# Embed text
text_embedding = await embedder.embed_text("description", media_item.id)

# Embed all aspects of media
embeddings = await embedder.embed_media(media_item)

# Compute similarity
similarity = await embedder.compute_similarity(embedding1, embedding2)
```

### MultimodalIndex

Stores and retrieves multimodal memories.

```python
from dominusprime.agents.memory.multimodal.index import MultimodalIndex
from dominusprime.agents.memory.multimodal.models import MediaType

index = MultimodalIndex(db_path=Path("./data/multimodal.db"), embedder=embedder)

# Store memory
memory = await index.store_memory(
    media_item=media_item,
    embeddings=embeddings,
    text_content="User error screenshot",
    experience_id="exp123"
)

# Get memory by ID
memory = await index.get_memory(media_id)

# Search memories
results = await index.search(
    query="error screenshot",
    media_type=MediaType.IMAGE,
    limit=10,
    similarity_threshold=0.5
)

for result in results:
    print(f"Similarity: {result.similarity_score}")
    print(f"Match Type: {result.match_type}")
    print(f"Description: {result.memory.media_item.description}")

# Delete memory
deleted = await index.delete_memory(media_id)
```

## Proactive Delivery System

### ContextMonitor

Monitors conversation context for triggers.

```python
from dominusprime.agents.memory.proactive.context_monitor import ContextMonitor

monitor = ContextMonitor(
    window_size=10,
    signal_threshold=0.3
)

# Add message and detect signals
signals = await monitor.add_message(user_message)

for signal in signals:
    print(f"Signal: {signal.signal_type}")
    print(f"Strength: {signal.strength}")
    print(f"Keywords: {signal.keywords}")

# Get context summary
context = monitor.get_recent_context()

# Check if good time to deliver
is_opportune = monitor.should_deliver_now()
```

### RelevanceScorer

Scores memory relevance to current context.

```python
from dominusprime.agents.memory.proactive.relevance_scorer import RelevanceScorer

scorer = RelevanceScorer(
    keyword_weight=0.3,
    topic_weight=0.25,
    temporal_weight=0.15,
    pattern_weight=0.2,
    context_weight=0.1
)

# Score memory
score = await scorer.score_memory(
    memory=memory_dict,
    signals=context_signals,
    current_context=context
)

print(f"Total Score: {score.total_score}")
print(f"Confidence: {score.confidence}")
print(f"Explanation: {score.explanation}")
```

### DeliveryManager

Manages proactive memory delivery.

```python
from dominusprime.agents.memory.proactive.delivery_manager import DeliveryManager
from dominusprime.agents.memory.proactive.models import DeliveryStrategy

manager = DeliveryManager(
    context_monitor=monitor,
    relevance_scorer=scorer,
    strategy=DeliveryStrategy.BALANCED,
    max_deliveries_per_session=10,
    min_confidence=0.6
)

# Evaluate candidate memories
proactive_memories = await manager.evaluate_memories(
    memories=candidate_memories,
    signals=signals,
    current_context=context
)

# Check if should deliver
for memory in proactive_memories:
    if await manager.should_deliver(memory):
        # Deliver memory
        result = await manager.deliver_memory(memory, current_message)
        print(result["formatted"])

# Get statistics
stats = manager.get_delivery_stats()

# Reset for new session
manager.reset_session()
```

## Database System

### DatabaseManager

Manages multiple SQLite databases.

```python
from dominusprime.database.connection import DatabaseManager

# Initialize
db_manager = DatabaseManager(data_dir=Path("./data"))
await db_manager.initialize()

# Get connection
conn = db_manager.get_connection("experiences")

# Execute with transaction
async with db_manager.transaction("experiences"):
    cursor = conn.cursor()
    cursor.execute("INSERT INTO experiences (...) VALUES (...)")

# Close all connections
await db_manager.close_all()
```

## Error Handling

### Security Errors

```python
from dominusprime.security.models import InterceptAction

try:
    result = await execute_shell_command_secure(command)
except PermissionError as e:
    # Command was blocked or rejected
    print(f"Access denied: {e}")
except TimeoutError as e:
    # Approval request timed out
    print(f"Approval timeout: {e}")
except RuntimeError as e:
    # Security manager not initialized
    print(f"System error: {e}")
```

### Database Errors

```python
import sqlite3

try:
    await db_manager.initialize()
except sqlite3.Error as e:
    print(f"Database error: {e}")
```

## Best Practices

### 1. Always Initialize Systems in Order

```python
# Initialize in this order
db_manager = DatabaseManager(data_dir=Path("./data"))
await db_manager.initialize()

security = SecurityManager.initialize(config, db_manager)

experience_system = ExperienceSystem(
    working_dir=Path("./working"),
    chat_model=your_chat_model,
    formatter=your_formatter,
    db_manager=db_manager
)
```

### 2. Use Async Context Managers

```python
async with db_manager.transaction("experiences"):
    # Your database operations
    pass
```

### 3. Handle Security Gracefully

```python
try:
    result = await execute_shell_command_secure(cmd)
except PermissionError:
    # Fallback or inform user
    result = None
```

### 4. Clean Up Resources

```python
# On shutdown
await db_manager.close_all()
```

### 5. Monitor Performance

```python
# Check delivery stats periodically
stats = manager.get_delivery_stats()
if stats["delivered_count"] > stats["max_deliveries"]:
    manager.reset_session()
```
