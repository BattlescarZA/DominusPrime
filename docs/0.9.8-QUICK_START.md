# DominusPrime 0.9.8 - Quick Start Guide

## Overview

DominusPrime 0.9.8 introduces powerful memory and security enhancements:

- 🧠 **Experience Distillation** - Learn from conversations automatically
- 🖼️ **Multimodal Memory** - Store and search images, audio, video
- 🎯 **Proactive Delivery** - Surface relevant memories at the right time
- 🔒 **Shell Security** - Risk analysis and approval for commands
- 🛡️ **Configurable Security** - 5 security levels from Open to Paranoid

## Installation

### Requirements

- **Python 3.10** (required for dependency compatibility)
- SQLite3 (included with Python)

### Install DominusPrime 0.9.8

```bash
# Checkout 0.9.8 branch
git checkout 0.9.8

# Run installer (automatically uses Python 3.10)
# Windows (PowerShell)
.\scripts\install.ps1

# Linux/macOS
./scripts/install.sh

# Windows (CMD)
scripts\install.bat
```

### Optional ML Dependencies

For full multimodal capabilities:

```bash
pip install pillow transformers sentence-transformers
```

## Quick Start

### 1. Initialize Data Directory

```bash
# Create data directory
mkdir -p data

# The system will auto-create databases on first use:
# - data/experiences.db (experience and skills storage)
# - data/multimodal.db (media items and embeddings)
# - data/security.db (audit logs and security events)
```

### 2. Configure Security (Optional)

```bash
# Set security profile
dominusprime security set-profile balanced

# Or create custom config
mkdir -p config
cat > config/security.json << EOF
{
  "security_level": "BALANCED",
  "shell_require_approval": true,
  "shell_block_dangerous": true,
  "audit_logging_enabled": true
}
EOF
```

### 3. Use in Your Agent

```python
from pathlib import Path
from dominusprime.database.connection import DatabaseManager
from dominusprime.security.manager import SecurityManager
from dominusprime.security.config import SecurityConfig
from dominusprime.agents.memory.experience.system import ExperienceSystem

# Initialize systems
async def initialize_agent(chat_model, formatter):
    # Database
    db_manager = DatabaseManager(Path("./data"))
    await db_manager.initialize()
    
    # Security
    config = SecurityConfig.from_file("./config/security.json")
    security = SecurityManager.initialize(config, db_manager)
    
    # Experience system (requires chat_model, formatter, and db_manager)
    experience_system = ExperienceSystem(
        working_dir=Path("./working"),
        chat_model=chat_model,      # ChatModelBase instance
        formatter=formatter,         # FormatterBase instance
        db_manager=db_manager       # DatabaseManager instance
    )
    
    return db_manager, security, experience_system
```

### 4. Use Secure Shell Commands

```python
from dominusprime.agents.tools.shell_secure import execute_shell_command_secure

# Execute with security checks
try:
    return_code, stdout, stderr = await execute_shell_command_secure(
        command="ls -la",
        session_id="session123"
    )
    print(f"Output: {stdout}")
except PermissionError as e:
    print(f"Command blocked: {e}")
```

### 5. Distill Experiences

```python
# After a conversation ends
await experience_system.distill_experiences(
    messages=conversation_messages,
    session_id="session123"
)

# Skills are automatically generated in working_dir/skills/learned/
```

### 6. Process Multimodal Content

```python
from dominusprime.agents.memory.multimodal.processor import MediaProcessor
from dominusprime.agents.memory.multimodal.embedder import MultimodalEmbedder
from dominusprime.agents.memory.multimodal.index import MultimodalIndex
from dominusprime.agents.memory.multimodal.models import MediaType

# Process image
processor = MediaProcessor(storage_dir=Path("./data/media"))
media_item = await processor.process_file(
    file_path="screenshot.png",
    session_id="session123",
    user_context="Error screenshot"
)

# Generate embeddings and store
embedder = MultimodalEmbedder()
embeddings = await embedder.embed_media(media_item)

index = MultimodalIndex(db_path=Path("./data/multimodal.db"), embedder=embedder)
await index.store_memory(media_item, embeddings, text_content="Python error screenshot")

# Search later
results = await index.search(
    query="python error",
    media_type=MediaType.IMAGE,
    limit=5
)
```

### 7. Proactive Memory Delivery

```python
from dominusprime.agents.memory.proactive import (
    ContextMonitor,
    RelevanceScorer,
    DeliveryManager,
    DeliveryStrategy
)

# Initialize proactive delivery
monitor = ContextMonitor(window_size=10)
scorer = RelevanceScorer()
manager = DeliveryManager(
    context_monitor=monitor,
    relevance_scorer=scorer,
    strategy=DeliveryStrategy.BALANCED
)

# Process incoming messages
signals = await monitor.add_message(user_message)

# Evaluate and deliver relevant memories
if signals:
    memories = await experience_system.search_experiences(...)
    proactive = await manager.evaluate_memories(memories, signals, monitor.get_recent_context())
    
    for mem in proactive:
        if await manager.should_deliver(mem):
            result = await manager.deliver_memory(mem, user_message)
            print(result["formatted"])
```

## Configuration

### Security Profiles

Choose a profile based on your needs:

- **open** - No restrictions (development only)
- **relaxed** - Warnings only, auto-approve most commands
- **balanced** - Standard protection (recommended)
- **strict** - Enhanced security, frequent approvals
- **paranoid** - Maximum security, approve everything

```bash
dominusprime security set-profile balanced
```

### Custom Configuration

Create `config/security.json`:

```json
{
  "security_level": "BALANCED",
  "shell_require_approval": true,
  "shell_block_dangerous": true,
  "tool_default_permission": "PROMPT",
  "skills_scan_on_load": true,
  "skills_require_approval": false,
  "audit_logging_enabled": true,
  "rate_limit_enabled": true,
  "max_commands_per_minute": 20
}
```

## Troubleshooting

### Security Manager Not Initialized

```python
# Make sure to initialize before use
from dominusprime.security.manager import SecurityManager
from dominusprime.security.config import SecurityConfig

config = SecurityConfig.from_file("./config/security.json")
security = SecurityManager.initialize(config, db_manager)
```

### ExperienceSystem Requires Dependencies

The [`ExperienceSystem`](../src/dominusprime/agents/memory/experience/system.py:23) class requires three parameters:

```python
from dominusprime.agents.memory.experience.system import ExperienceSystem

# CORRECT: Provide all required dependencies
experience_system = ExperienceSystem(
    working_dir=Path("./working"),
    chat_model=your_chat_model,    # ChatModelBase instance (required)
    formatter=your_formatter,       # FormatterBase instance (required)
    db_manager=db_manager          # DatabaseManager instance (required)
)

# INCORRECT: Missing required parameters will cause errors
# experience_system = ExperienceSystem(working_dir=Path("./working"))
```

### Missing ML Dependencies

The system gracefully degrades without optional dependencies:
- No Pillow: Basic file info only (no image processing)
- No transformers: Fallback embeddings used
- No sentence-transformers: Hash-based embeddings

Install for full functionality:
```bash
pip install pillow transformers sentence-transformers
```

### Database Locked

```python
# Ensure proper async context management
async with db_manager.transaction("experiences"):
    # Your database operations
    pass
```

### Python Version Issues

If you encounter dependency errors:
- Ensure Python 3.10 is installed (not 3.12)
- Re-run the installer: `.\scripts\install.ps1` or `./scripts/install.sh`
- The installer automatically uses Python 3.10

## Complete Example

Here's a complete example integrating all features:

```python
from pathlib import Path
from dominusprime.database.connection import DatabaseManager
from dominusprime.security.manager import SecurityManager
from dominusprime.security.config import SecurityConfig
from dominusprime.agents.memory.experience.system import ExperienceSystem
from dominusprime.agents.memory.proactive import DeliveryManager, ContextMonitor, RelevanceScorer
from dominusprime.agents.tools.shell_secure import execute_shell_command_secure

class EnhancedAgent:
    def __init__(self, chat_model, formatter):
        self.chat_model = chat_model
        self.formatter = formatter
        
    async def initialize(self):
        # Initialize database
        self.db_manager = DatabaseManager(Path("./data"))
        await self.db_manager.initialize()
        
        # Initialize security
        config = SecurityConfig.from_file("./config/security.json")
        self.security = SecurityManager.initialize(config, self.db_manager)
        
        # Initialize experience system
        self.experience_system = ExperienceSystem(
            working_dir=Path("./working"),
            chat_model=self.chat_model,
            formatter=self.formatter,
            db_manager=self.db_manager
        )
        
        # Initialize proactive delivery
        self.monitor = ContextMonitor(window_size=10)
        self.scorer = RelevanceScorer()
        self.delivery_manager = DeliveryManager(
            context_monitor=self.monitor,
            relevance_scorer=self.scorer
        )
    
    async def process_message(self, message):
        # Monitor context
        signals = await self.monitor.add_message(message)
        
        # Deliver proactive memories
        if signals:
            memories = await self.experience_system.search_experiences(message.content)
            proactive = await self.delivery_manager.evaluate_memories(
                memories, signals, self.monitor.get_recent_context()
            )
            
            for mem in proactive:
                if await self.delivery_manager.should_deliver(mem):
                    await self.delivery_manager.deliver_memory(mem, message)
        
        # Process normally
        response = await self.generate_response(message)
        return response
    
    async def execute_command(self, command):
        # Secure execution
        try:
            return await execute_shell_command_secure(
                command=command,
                session_id=self.session_id
            )
        except PermissionError as e:
            return f"Command blocked: {e}"
    
    async def end_conversation(self, messages):
        # Distill experiences
        await self.experience_system.distill_experiences(
            messages=messages,
            session_id=self.session_id
        )

# Usage
agent = EnhancedAgent(chat_model, formatter)
await agent.initialize()
response = await agent.process_message(user_message)
```
